  export async function findContacts(
    company: string,
    role: string,
    location: string | null,
  ): Promise<ContactResult[]> {
    const userMessage = `Find 3 contacts at ${company} relevant to the role: ${role}. Location: ${location ?? "any"}. Return JSON only.`;

    const rawOutput = await runAgentLoop({
      systemPrompt: CONTACT_FINDER_SYSTEM_PROMPT,
      userMessage,
      tools: [
        { type: "web_search_20250305" as any, name: "web_search", max_uses: 5 },
      ],
      executeTool: async () => "", // Server tools handled by API
      maxSteps: 5,
      maxTokens: 2048,
    });

    // Parse and validate JSON output
    const parsed = extractJsonArray(rawOutput);
    const validated = parsed.map(mapToContactResult);

    // D-03: Always return exactly 3
    while (validated.length < 3) {
      validated.push({
        name: "Unknown Contact",
        title: "Unknown",
        sourceUrl: null,
        confidence: "low",
        publicActivity: null,
      });
    }

    return validated.slice(0, 3);
  }
