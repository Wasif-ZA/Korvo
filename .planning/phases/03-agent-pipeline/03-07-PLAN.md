- Export `draftEmails` function:
```typescript
interface DraftInput {
  contact: ContactResult;
  research: ResearchCard;
  score: ScoreResult;
  targetRole: string;
  userName: string; // The user's name for the email sign-off
}

export async function draftEmails(
  inputs: DraftInput[]
): Promise<DraftResult[]> {
  const results: DraftResult[] = [];

  for (const { contact, research, score, targetRole, userName } of inputs) {
    try {
      const systemPrompt = buildDrafterSystemPrompt(score.tone);

      const userMessage = `Draft a cold email to ${contact.name} (${contact.title}).

Research:
- Background: ${research.background}
- Mention: ${research.mentionThis}
- Hooks: ${research.hooks.join(", ") || "none found"}

Score: ${score.total}/100 (tone: ${score.tone})
Target role: ${targetRole}
My name: ${userName}

Return JSON only.`;

      const rawOutput = await runAgentLoop({
        systemPrompt,
        userMessage,
        tools: [], // No tools needed — drafter works from provided context
        executeTool: async () => "",
        maxSteps: 2,
        maxTokens: 1024,
      });

      const parsed = extractJsonObject(rawOutput);

      // Validate and sanitize
      let body = parsed.body ?? "";
      // Remove em dashes if Claude includes them despite instructions
      body = body.replace(/\u2014/g, ",").replace(/--/g, ",");

      results.push({
        subject: parsed.subject ?? `Connecting about ${targetRole} at ${contact.title.split(" ").pop()}`,
        body,
        templateType: validateTemplateType(parsed.template_type),
        hookUsed: parsed.hook_used ?? research.hooks[0] ?? "",
      });
    } catch {
      // D-03: Never fail — generate a safe fallback
      results.push(generateFallbackDraft(contact, research, score, targetRole, userName));
    }
  }

  return results;
}
````
