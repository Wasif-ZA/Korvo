    1. Create `lib/stripe/webhooks.ts` — webhook event handler functions:

    ```typescript
    import { prisma } from '@/lib/db/prisma'
    import { stripe } from '@/lib/stripe/client'
    import type Stripe from 'stripe'

    export async function handleCheckoutCompleted(session: Stripe.Checkout.Session) {
      // Retrieve full session with expanded data (Pitfall 7)
      const fullSession = await stripe.checkout.sessions.retrieve(session.id, {
        expand: ['subscription'],
      })

      const userId = fullSession.metadata?.user_id
      if (!userId) {
        console.error('Checkout session missing user_id in metadata')
        return
      }

      const subscription = fullSession.subscription as Stripe.Subscription

      // Update profile — using Prisma directly (service role not needed with Prisma, it bypasses RLS)
      await prisma.profile.update({
        where: { userId },
        data: {
          plan: 'pro',
          stripeCustomerId: fullSession.customer as string,
          stripeSubscriptionId: subscription.id,
        },
      })
    }

    export async function handleSubscriptionDeleted(subscription: Stripe.Subscription) {
      // Find profile by stripeSubscriptionId and downgrade
      await prisma.profile.updateMany({
        where: { stripeSubscriptionId: subscription.id },
        data: {
          plan: 'free',
          stripeSubscriptionId: null,
        },
      })
    }
    ```

    Per PAY-04: Plan stored in profiles table, updated by server-side webhook handler only.
    Per D-08: Data is kept on subscription end — user just returns to free tier limits.

    2. Create `app/api/stripe/webhooks/route.ts`:
    ```typescript
    import { NextRequest, NextResponse } from 'next/server'
    import Stripe from 'stripe'
    import { stripe } from '@/lib/stripe/client'
    import { handleCheckoutCompleted, handleSubscriptionDeleted } from '@/lib/stripe/webhooks'

    export async function POST(req: NextRequest) {
      // CRITICAL: Use req.text() NOT req.json() — Pitfall 2
      const rawBody = await req.text()
      const signature = req.headers.get('stripe-signature')

      if (!signature) {
        return NextResponse.json({ error: 'Missing stripe-signature header' }, { status: 400 })
      }

      let event: Stripe.Event
      try {
        event = stripe.webhooks.constructEvent(
          rawBody,
          signature,
          process.env.STRIPE_WEBHOOK_SECRET!
        )
      } catch (err) {
        console.error('Webhook signature verification failed:', err)
        return NextResponse.json({ error: 'Webhook signature verification failed' }, { status: 400 })
      }

      switch (event.type) {
        case 'checkout.session.completed': {
          const session = event.data.object as Stripe.Checkout.Session
          await handleCheckoutCompleted(session)
          break
        }
        case 'customer.subscription.deleted': {
          const subscription = event.data.object as Stripe.Subscription
          await handleSubscriptionDeleted(subscription)
          break
        }
        default:
          // Unhandled event type — log and acknowledge
          console.log(`Unhandled webhook event type: ${event.type}`)
      }

      return NextResponse.json({ received: true })
    }
    ```

    CRITICAL NOTES:
    - Line 1 of POST handler MUST be `const rawBody = await req.text()` — NOT `req.json()` (Pitfall 2)
    - This route is excluded from proxy.ts auth guard (no auth required for webhooks)
    - Prisma client connects via service role equivalent (Prisma bypasses RLS since it connects directly to PostgreSQL, not through Supabase PostgREST)
    - Per Pitfall 7: Always retrieve full session via `stripe.checkout.sessions.retrieve()` with expand
