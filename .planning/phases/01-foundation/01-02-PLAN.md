  <action>
    1. Generate migration without applying: `npx prisma migrate dev --create-only --name create_core_tables`

    2. Open the generated migration SQL file and APPEND the following RLS statements after the Prisma-generated CREATE TABLE statements:

    ```sql
    -- ============================================
    -- ROW LEVEL SECURITY (per D-14, D-22, FOUND-04)
    -- CRITICAL: Gate on profiles table, NEVER user_metadata
    -- ============================================

    -- Enable RLS on all user-scoped tables
    ALTER TABLE "profiles" ENABLE ROW LEVEL SECURITY;
    ALTER TABLE "searches" ENABLE ROW LEVEL SECURITY;
    ALTER TABLE "contacts" ENABLE ROW LEVEL SECURITY;
    ALTER TABLE "outreach" ENABLE ROW LEVEL SECURITY;
    ALTER TABLE "guest_ip_limits" ENABLE ROW LEVEL SECURITY;

    -- profiles: users can only read/update their own profile
    CREATE POLICY "profiles_select_own" ON "profiles"
      FOR SELECT USING (auth.uid() = user_id);
    CREATE POLICY "profiles_update_own" ON "profiles"
      FOR UPDATE USING (auth.uid() = user_id);
    -- INSERT handled by trigger on auth.users creation (Supabase manages)
    -- Service role key bypasses RLS for webhook handler updates

    -- searches: scoped to owner, guests can insert with null user_id
    CREATE POLICY "searches_select_own" ON "searches"
      FOR SELECT USING (auth.uid() = user_id);
    CREATE POLICY "searches_insert" ON "searches"
      FOR INSERT WITH CHECK (auth.uid() = user_id OR user_id IS NULL);
    CREATE POLICY "searches_update_own" ON "searches"
      FOR UPDATE USING (auth.uid() = user_id);

    -- contacts: scoped via search ownership
    CREATE POLICY "contacts_select_own" ON "contacts"
      FOR SELECT USING (
        auth.uid() = (SELECT user_id FROM searches WHERE id = search_id)
      );
    CREATE POLICY "contacts_insert_own" ON "contacts"
      FOR INSERT WITH CHECK (
        auth.uid() = (SELECT user_id FROM searches WHERE id = search_id)
        OR (SELECT user_id FROM searches WHERE id = search_id) IS NULL
      );
    CREATE POLICY "contacts_update_own" ON "contacts"
      FOR UPDATE USING (
        auth.uid() = (SELECT user_id FROM searches WHERE id = search_id)
      );

    -- outreach: scoped via contact -> search ownership
    CREATE POLICY "outreach_select_own" ON "outreach"
      FOR SELECT USING (
        auth.uid() = (
          SELECT s.user_id FROM contacts c
          JOIN searches s ON s.id = c.search_id
          WHERE c.id = contact_id
        )
      );
    CREATE POLICY "outreach_insert_own" ON "outreach"
      FOR INSERT WITH CHECK (
        auth.uid() = (
          SELECT s.user_id FROM contacts c
          JOIN searches s ON s.id = c.search_id
          WHERE c.id = contact_id
        )
      );
    CREATE POLICY "outreach_update_own" ON "outreach"
      FOR UPDATE USING (
        auth.uid() = (
          SELECT s.user_id FROM contacts c
          JOIN searches s ON s.id = c.search_id
          WHERE c.id = contact_id
        )
      );

    -- guest_ip_limits: no user-scoped access needed (server-side only via service role)
    -- No SELECT/INSERT/UPDATE policies for anon — only service role accesses this table

    -- ============================================
    -- PROFILE AUTO-CREATION TRIGGER
    -- Creates a profile row when a new user signs up via Supabase Auth
    -- ============================================
    CREATE OR REPLACE FUNCTION public.handle_new_user()
    RETURNS trigger AS $$
    BEGIN
      INSERT INTO public.profiles (id, user_id, email, full_name, avatar_url, plan, searches_used_this_month, drafts_used_this_month, searches_reset_at, created_at, updated_at)
      VALUES (
        gen_random_uuid(),
        NEW.id,
        NEW.email,
        NEW.raw_user_meta_data->>'full_name',
        NEW.raw_user_meta_data->>'avatar_url',
        'free',
        0,
        0,
        NOW(),
        NOW(),
        NOW()
      );
      RETURN NEW;
    END;
    $$ LANGUAGE plpgsql SECURITY DEFINER;

    CREATE OR REPLACE TRIGGER on_auth_user_created
      AFTER INSERT ON auth.users
      FOR EACH ROW EXECUTE FUNCTION public.handle_new_user();
    ```

    3. Apply the migration: `npx prisma migrate dev`

    - Per D-14/D-22: RLS is version-controlled in migration files, not dashboard-only.
    - The service role key bypasses RLS automatically — no explicit policy needed for the webhook handler.
