     // Optimistic: update local data immediately
     mutateContacts(
       (current: any) => ({
         ...current,
         data: current?.data?.map((c: any) =>
           c.id === id ? { ...c, stage: newStage } : c
         ),
       }),
       { revalidate: false }
     );

     try {
       const res = await fetch(`/api/contacts/${id}`, {
         method: "PATCH",
         headers: { "Content-Type": "application/json" },
         body: JSON.stringify({ stage: newStage }),
       });
       if (!res.ok) throw new Error("Failed");
       toast.success(`Moved to ${formatStage(newStage)}`);
       mutateContacts(); // revalidate for consistency
     } catch {
       mutateContacts(); // revert optimistic update
       toast.error("Couldn't update stage. Please try again.");
     }
